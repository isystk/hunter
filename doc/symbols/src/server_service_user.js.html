 18 /**
 19  * オートコンプリート用サーチを実行します
 20  *
 21  * @param data
 22  * @param callback
 23  * @return
 24  */
 25 UserService.prototype.autocomplete = function(data, callback) {
 26 
 27 	var colname = 'users';
 28 	var term = data.term ? data.term : '';
 29 	var size = data.size ? data.size : 10;
 30 	var reg = data.reg ? data.reg : '';
 31 
 32 	var field = 'screenName';
 33 	if (data.field) {
 34 		field = data.field;
 35 	}
 36 
 37 	if (data.reg == 'forward') { // 前方一致
 38 		reg = new RegExp('^' + term + '.*');
 39 	} else if (data.reg == 'exact') { // 完全一致
 40 		reg = term;
 41 	} else {
 42 		reg = new RegExp('.*' + term + '.*', 'i');
 43 	}
 44 
 45 	var query = {};
 46 	query[field] = reg;
 47 
 48 	var result = { list: [] };
 49 	commonDao.list(colname, {where: query, limit: size}, function(err, result) {
 50 		var list = [];
 51 		for (var i=0,len=result.length; i<len; i++) {
 52 			var item = result[i];
 53 			list.push({
 54 				label: item.screenName + ' (' + item.name + ')',
 55 				value: item._id
 56 			 });
 57 		}
 58 		callback(null, list);
 59 	});
 60 };

122 /**
123  * プロフィールを取得します。
124  * @param {Object} data
125  * @param {function} callback
126  */
127 UserService.prototype.getProfile = function(data, callback) {
128 
129 	next(function() {
130 		var deferred = new Deferred();
131 		// ユーザー情報を取得します
132 		commonDao.find('users', {_id: data.id}, function(err, result) {
133 			deferred.call(result);
134 		});
135 		return deferred;
136 	}).next(function(user) {
137 		// ユーザーの合計ポイントから現在のレベルを取得します
138 		commonDao.list('levels', {where: {totalScore: {$lte: user.totalScore || 0}}, sort: {level: -1}, limit: 1}, function(err, results) {
139 			var result = results[0];
140 			var level = result.level;
141 
142 			var profile = {
143 				id: user.id,
144 				screenName: user.screenName || '',
145 				name: user.name || '',
146 				image: user.image,
147 				totalScore: user.totalScore || 0,
148 				highScore: user.highScore || 0,
149 				level: level || 0
150 			};
151 			callback(err, profile);
152 		});
153 	});
154 };
