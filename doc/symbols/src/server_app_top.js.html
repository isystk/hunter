 15 /**
 16  * サインイン処理 この処理は９leapからのアクセスようです。
 17  * @param {Object} user
 18  * @param {Object} data
 19  */
 20 TopControl.prototype.signin = function(user, data) {
 21 	next(function() {
 22 		var deferred = new Deferred();
 23 		// ユーザー情報を登録します
 24 		userService.registProfile({profile: data.profile}, function(err, result) {
 25 			deferred.call(result);
 26 		});
 27 		return deferred;
 28 	}).next(function(data) {
 29 		// ユーザー情報を取得します
 30 		userService.getProfileByUserId({snsType: data.snsType, userId: data.userId}, function(err, result) {
 31 			user.send('top.signin', {id: result._id});
 32 		});
 33 	});
 34 };

 36 /**
 37  * 初期表示処理
 38  * @param {Object} user
 39  * @param {Object} data
 40  */
 41 TopControl.prototype.init = function(user, data) {
 42 	next(function() {
 43 		var memberList = applications.getMemberList();
 44 		// マップ上に表示するゲーム情報を取得します。
 45 		gameService.getGameData({memberList: memberList}, function(err, result) {
 46 			user.send('game.init', {actors: result.actors, items: result.items});
 47 		});
 48 	}).next(function() {
 49 		// 現在接続中のメンバー一覧を取得します。
 50 		var memberList = applications.getMemberList();
 51 		user.send('top.memberList', {memberList:memberList});
 52 	}).next(function() {
 53 	});
 54 };

 56 /**
 57  * ログイン処理
 58  * @param {Object} user
 59  * @param {Object} data
 60  */
 61 TopControl.prototype.login = function(user, data) {
 62 
 63 	next(function() {
 64 		var deferred = new Deferred();
 65 		// ユーザー情報を取得します
 66 		userService.getProfile({id: data.id}, function(err, result) {
 67 			deferred.call(result);
 68 		});
 69 		return deferred;
 70 	}).next(function(profile) {
 71 
 72 		user.id = profile.id;
 73 		user.screenName = profile.screenName;
 74 		user.name = profile.name;
 75 		user.image = profile.image;
 76 		user.game = {};
 77 		user.game.totalScore = profile.totalScore;
 78 		user.game.highScore = profile.highScore;
 79 		user.game.level = profile.level;
 80 
 81 		// ユーザー情報をアプリケーションに追加します。
 82 		applications.putUser(user);
 83 
 84 		// メンバーリストを送信
 85 		var memberList = applications.getMemberList();
 86 		user.send('top.memberList', {memberList:memberList});
 87 		user.sendAll('top.memberList', {memberList:memberList});
 88 
 89 		user.send('top.login', {id: user.id, name: user.name, image: user.image});
 90 		user.send('top.message', {message: user.name +'さん。こんにちわ。'});
 91 
 92 		// ユーザーログ
 93 		userService.log({'type': LOG_TYPE.LOGIN.key, userId: user.id, data:{userName: user.name}});
 94 	}).next(function() {
 95 		// 現在接続中のメンバー一覧を取得します。
 96 		var memberList = applications.getMemberList();
 97 		user.send('top.memberList', {memberList:memberList});
 98 	});
 99 };
