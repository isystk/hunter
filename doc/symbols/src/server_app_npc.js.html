 17 /**
 18  * モンスターを生成します
 19  */
 20 Npc.prototype.createNpc = function() {
 21 	var self = this;
 22 
 23 	next(function() {
 24 		// NPCを配置する位置をランダムに生成します。
 25 		var ramPos = function () {
 26 			var x = Math.floor( Math.random() * 40 ) * 16; // 初期ポジション
 27 			var y = Math.floor( Math.random() * 30 ) * 16; // 初期ポジション
 28 			// 生成した位置が障害物上の場合は、生成し直す
 29 			if (hitTest(x, y)) {
 30 				return ramPos();
 31 			}
 32 			return {x: x, y: y};
 33 		};
 34 
 35 		// アイテムを生成する
 36 		var createItem = function() {
 37 			var t = setTimeout(function() {
 38 				// アイテムが無くなった場合は、追加する
 39 				if (_.isEmpty(self.items)) {
 40 					var pos = ramPos();
 41 					var id = new Date().getTime();
 42 					self.items[id] = {id: id, itemType: Const.ItemType.STAR,  name: '',  x: pos.x, y: pos.y};
 43 					var users = applications.getUserList();
 44 					for (var key in users) {
 45 						var user = users[key];
 46 						user.send('game.addItem', {items: self.items});
 47 					}
 48 				}
 49 				createItem();
 50 			}, 3000);
 51 		};
 52 		createItem();
 53 	}).next(function() {
 54 		// NPCを配置する位置をランダムに生成します。
 55 		var ramPos = function () {
 56 			var x = Math.floor( Math.random() * 40 ) * 16 - 8; // 初期ポジション
 57 			var y = Math.floor( Math.random() * 30 ) * 16; // 初期ポジション
 58 			// 生成した位置が障害物上の場合は、生成し直す
 59 			if (hitTest(x, y)) {
 60 				return ramPos();
 61 			}
 62 			return {x: x, y: y};
 63 		};
 64 
 65 		// NPCを生成する
 66 		var deferred = new Deferred();
 67 		commonDao.list('npcs', {}, function(err, result) {
 68 			for (var i=0, len=result.length; i<len; i++) {
 69 				var data = result[i];
 70 				var pos = ramPos();
 71 				var monster = {
 72 					id: data._id,
 73 					//name: data.name || '',
 74 					name: '',
 75 					actorType: data.actorType,
 76 					x: pos.x,
 77 					y: pos.y,
 78 					talk: ''
 79 				};
 80 				self.npcs[monster.id] = monster;
 81 			}
 82 			deferred.call();
 83 		});
 84 		return deferred;
 85 	}).next(function() {
 86 		var MOVE_TYPE = {0: 'right', 1: 'up', 2: 'left', 3: 'down'};
 87 		var moveNpc = function() {
 88 			var t = setTimeout(function() {
 89 				for (var key in self.npcs) {
 90 					var npc = self.npcs[key];
 91 					var pos = MOVE_TYPE[Math.floor( Math.random() * 4 )];
 92 					var x = npc.x;
 93 					var y = npc.y;
 94 					if (pos === 'right') {
 95 						x = x + 16;
 96 					}
 97 					else if (pos === 'up') {
 98 						y = y + 16;
 99 					}
100 					else if (pos === 'left') {
101 						x = x - 16;
102 					}
103 					else if (pos === 'down') {
104 						y = y - 16;
105 					}
106 					// 衝突判定
107 					var map = {};
108 					map.width = 640;
109 					map.height = 480;
110 					if (x <= 0 || map.width < x || y <= 0 || map.height < y || hitTest(x, y)) {
111 						continue;
112 					}
113 
114 					npc.x = x;
115 					npc.y = y;
116 
117 					self.npcs[key] = npc;
118 					clearInterval(t);
119 				}
120 				var users = applications.getUserList();
121 				for (var key in users) {
122 					var user = users[key];
123 					user.send('game.actorMove', {actors: self.npcs});
124 				}
125 				moveNpc();
126 			}, 3000);
127 		};
128 		moveNpc();
129 	});
130 };

153 /**
154  * Map上に障害物があるかどうかを判定する.
155  * @param {Number} x 判定を行うマップ上の点のx座標.
156  * @param {Number} y 判定を行うマップ上の点のy座標.
157  * @return {Boolean} 障害物があるかどうか.
158  */
159 function hitTest (x, y) {
160 	var width = 640;
161 	var height = 480;
162 	var _image = {};
163 	_image.width = 256;
164 	_image.height = 256;
165 	var _tileWidth = 16;
166 	var _tileHeight = 16;
167 	var _data = [gameMap.backgroundMap, gameMap.fourgroundMap];
168 	var collisionData = gameMap.collisionData;
169 
170 	if (x < 0 || width <= x || y <= 16 || height <= y) {
171 		return true;
172 	}
173 	var width = _image.width;
174 	var height = _image.height;
175 	var tileWidth = _tileWidth || width;
176 	var tileHeight = _tileHeight || height;
177 	x = x / tileWidth | 0;
178 	y = y / tileHeight -1 | 0;
179 	if (collisionData != null) {
180 		return collisionData[y] && !!collisionData[y][x];
181 	} else {
182 		for (var i = 0, len = _data.length; i < len; i++) {
183 			var data = _data[i];
184 			var n;
185 			if (data[y] != null && (n = data[y][x]) != null &&
186 				0 <= n && n < (width / tileWidth | 0) * (height / tileHeight | 0)) {
187 				return true;
188 			}
189 		}
190 		return false;
191 	}
192 };
