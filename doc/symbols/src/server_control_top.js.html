1 var TopService = require('../service/top');
  2 var GameService = require('../service/game');
  3 var Applications = require('../applications');
  4 var Deferred = require('jsdeferred').Deferred;
  5 Deferred.define();
  6 
  7 function TopControl() {
  8 }
  9 
 10 module.exports = new TopControl();
 11 
 12 /**
 13  * サインイン処理 この処理は９leapからのアクセスようです。
 14  * @param {Object} user
 15  * @param {Object} data
 16  */
 17 TopControl.prototype.signin = function(user, data) {
 18 	next(function() {
 19 		var deferred = new Deferred();
 20 		// ユーザー情報を登録します
 21 		TopService.registProfile({profile: data.profile}, function(err, result) {
 22 			deferred.call(result);
 23 		});
 24 		return deferred;
 25 	}).next(function(data) {
 26 		// ユーザー情報を取得します
 27 		TopService.getProfileByUserId({snsType: data.snsType, userId: data.userId}, function(err, result) {
 28 			user.send('top.signin', {id: result._id});
 29 		});
 30 	});
 31 };

 33 /**
 34  * 初期表示処理
 35  * @param {Object} user
 36  * @param {Object} data
 37  */
 38 TopControl.prototype.init = function(user, data) {
 39 	next(function() {
 40 		// マップ上に表示するゲーム情報を取得します。
 41 		GameService.getGameData({}, function(err, result) {
 42 			user.send('game.init', {actors: result.actors, items: result.items});
 43 		});
 44 	}).next(function() {
 45 		// 現在接続中のメンバー一覧を取得します。
 46 		var memberList = Applications.getMemberList();
 47 		user.send('top.memberList', {memberList:memberList});
 48 	}).next(function() {
 49 	});
 50 };

 52 /**
 53  * ログイン処理
 54  * @param {Object} user
 55  * @param {Object} data
 56  */
 57 TopControl.prototype.login = function(user, data) {
 58 
 59 	next(function() {
 60 		var deferred = new Deferred();
 61 		// ユーザー情報を取得します
 62 		TopService.getProfile({id: data.id}, function(err, result) {
 63 			deferred.call(result);
 64 		});
 65 		return deferred;
 66 	}).next(function(profile) {
 67 
 68 		user.id = profile.id;
 69 		user.screenName = profile.screenName;
 70 		user.name = profile.name;
 71 		user.image = profile.image;
 72 		user.game.totalScore = profile.totalScore;
 73 		user.game.highScore = profile.highScore;
 74 		user.game.level = profile.level;
 75 
 76 		// ユーザー情報をアプリケーションに追加します。
 77 		Applications.putUser(user);
 78 
 79 		user.send('top.login', {id: user.id, name: user.name, image: user.image});
 80 		user.send('top.message', {message: user.name +'さん。こんにちわ。'});
 81 	}).next(function() {
 82 		// 現在接続中のメンバー一覧を取得します。
 83 		var memberList = Applications.getMemberList();
 84 		user.send('top.memberList', {memberList:memberList});
 85 	});
 86 };

 88 /**
 89  * ランキング情報を取得します。
 90  * @param {Object} user
 91  * @param {Object} data
 92  */
 93 TopControl.prototype.getRanking = function(user, data) {
 94 	next(function() {
 95 		var deferred = new Deferred();
 96 		// ユーザー情報を取得します
 97 		TopService.getRanking({index: data.index}, function(err, result) {
 98 			deferred.call(result);
 99 		});
100 		return deferred;
101 	}).next(function(result) {
102 		user.send('top.getRanking', {ranking: result.ranking, max: result.max});
103 	});
104 };
