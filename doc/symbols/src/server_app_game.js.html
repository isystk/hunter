 15 /**
 16  * プレイヤーがゲームに参加
 17  * @param {Object} user
 18  * @param {Object} data
 19  */
 20 GameControl.prototype.actorJoin = function(user, data) {
 21 
 22 	// ゲームに既に参加済みの場合は以降処理しない
 23 	if (user.game.isJoin === true) {
 24 		return;
 25 	}
 26 
 27 	// ゲームに参加中に設定
 28 	user.game.isJoin = true;
 29 	user.game.actorType = Const.ActorType.HUMAN;
 30 	user.game.x = 6 * 16 - 8; // 初期ポジション
 31 	user.game.y = 10 * 16; // 初期ポジション
 32 
 33 	user.game.score = 0; // スコアを初期化
 34 
 35 	// ユーザー情報を更新
 36 	applications.putUser(user);
 37 
 38 	var actor = {
 39 		id: user.id,
 40 		name: user.name,
 41 		actorType: user.game.actorType,
 42 		x: user.game.x,
 43 		y: user.game.y,
 44 		totalScore: user.game.totalScore,
 45 		highScore: user.game.highScore,
 46 		level: user.game.level
 47 	};
 48 	user.send('game.actorJoin', {actor: actor});
 49 
 50 	var actors = {};
 51 	actors[user.id] = actor;
 52 	user.sendAll('game.actorAdd', {actors: actors});
 53 	user.sendAll('top.message', {message: user.name +'さんが、参加しました。'});
 54 
 55 	// ユーザーログ
 56 	userService.log({'type': LOG_TYPE.GAME_JOIN.key, userId: user.id, data:{totalScore: user.game.totalScore, highScore: user.game.highScore, level: user.game.level}});
 57 };

134 /**
135  * プレイヤーがアイテムを取得した
136  * @param {Object} user
137  * @param {Object} data
138  */
139 GameControl.prototype.getItem = function(user, data) {
140 
141 	// ゲームに不参加の場合は以降処理しない
142 	if (user.game.isJoin === false) {
143 		return;
144 	}
145 
146 	var itemId = data.itemId;
147 
148 	var point = 1;
149 	var score = user.game.score + point;
150 	var totalScore = user.game.totalScore + point;
151 	user.game.score = score;
152 	user.game.totalScore = totalScore;
153 
154 	// ハイスコア可動化を判定
155 	if (user.game.highScore < score) {
156 		user.game.highScore = score;
157 	}
158 
159 	// ユーザー情報を更新
160 	applications.putUser(user);
161 
162 	var result = {
163 		score: score,
164 		totalScore: totalScore
165 	};
166 	user.send('game.getItem', result);
167 
168 	var result2 = {
169 		itemId: itemId
170 	};
171 	user.send('game.removeItem', result2);
172 	user.sendAll('game.removeItem', result2);
173 
174 	gameService.getItem({id: user.id, totalScore: totalScore, highScore: user.game.highScore, itemId: itemId}, function(err, result) {
175 	});
176 
177 	// ユーザーログ
178 	userService.log({'type': LOG_TYPE.GAME_STAR.key, userId: user.id, data:{totalScore: user.game.totalScore, highScore: user.game.highScore, level: user.game.level}});
179 };
